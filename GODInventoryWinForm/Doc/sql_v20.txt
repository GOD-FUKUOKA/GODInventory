# 公司列表， 添加公司，删除公司，修改公司

  `id` INT NOT NULL AUTO_INCREMENT,
  `fullname` VARCHAR(45) NOT NULL,
  `parent_id` INT NOT NULL DEFAULT 0,
  `address` VARCHAR(128) NULL,
  `phone` VARCHAR(45) NULL,

  PRIMARY KEY (`id`));
